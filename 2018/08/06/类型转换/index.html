}

题目

// 题目1：如下代码输出什么?

if ("hello") {
  console.log("hello")
}

// 题目2：如下代码输出什么?

if ("") {
  console.log('empty')
}

// 题目3：如下代码输出什么?

if (" ") {
  console.log('blank')
}

// 题目4：如下代码输出什么?

if ([0]) {
  console.log('array')
}

if('0.00'){
  console.log('0.00')
}
